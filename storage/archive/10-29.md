```html
<!-- OLD PATTERN (margin-top creates gap OUTSIDE hero) -->
<section class="py-5 mt-5" style="background: gradient">
  <div class="container">
    <!-- content here -->
  </div>
</section>

<!-- NEW PATTERN (hero extends to top, gap is INSIDE as padding) -->
<section class="py-5" style="background: gradient">
  <div class="container" style="padding-top: 100px;">

  </div>
</section>
```
